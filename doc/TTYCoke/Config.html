def initialize
  log_rescue(self, __method__, caller)  {
    @files = {
      home: ENV['HOME'] + "/.ttycoke.yaml",
      tty_coke: File.dirname(__FILE__) + '/../../config/config.yaml'
    }
    begin
      @config = YAML::load(File.open(File.expand_path(file)))
    rescue TypeError => e
      raise Errors::TTYCokeTypeError.new(e)
    end
  }
end

def file f=nil
  log_rescue(self, __method__, caller)  {
    return f if f
    if FileTest.exist?(@files.fetch(:home))
      @files.fetch(:home)
    else
      @files.fetch(:tty_coke)
    end
  }
end
